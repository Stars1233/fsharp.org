    // A function generating a sequence of numbers
    let rec fizzBuzzSeq n = seq {
        yield
            match n with
            | x when x % 15 = 0 -> "fizzbuzz"
            | x when x % 3 = 0 -> "fizz"
            | x when x % 5 = 0 -> "buzz"
            | _ -> n.ToString()

        yield! fizzBuzzSeq (n + 1)
    }

    // Process the sequence using a pipeline
    fizzBuzzSeq 1
    |> Seq.take 100
    |> Seq.iter (printfn "%s")
